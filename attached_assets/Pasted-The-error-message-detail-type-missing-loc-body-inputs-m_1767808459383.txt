{

Complete Example for Podcast Generation
Here's a complete example of how to properly use the Text to Dialogue API for podcast generation:

import requests
ELEVENLABS_API_KEY = "your-api-key"
url = "https://api.elevenlabs.io/v1/text-to-dialogue"
headers = {
  "Accept": "audio/mpeg",
  "Content-Type": "application/json",
  "xi-api-key": ELEVENLABS_API_KEY
}
data = {
  "inputs": [
    {
      "text": "[enthusiastic] Welcome to our podcast! Today we're discussing AI advancements.",
      "voice_id": "first_speaker_voice_id"
    },
    {
      "text": "[curious] That's fascinating! Can you tell us more about recent developments?",
      "voice_id": "second_speaker_voice_id"
    },
    {
      "text": "[informative] Certainly! The latest models have shown remarkable improvements in understanding context.",
      "voice_id": "first_speaker_voice_id"
    }
  ],
  "model_id": "eleven_v3"
}
response = requests.post(url, json=data, headers=headers)
if response.status_code == 200:
    with open("podcast_dialogue.mp3", "wb") as f:
        f.write(response.content)
    print("Podcast audio saved as podcast_dialogue.mp3")
else:
    print(f"Error: {response.status_code}")
    print(response.text)
